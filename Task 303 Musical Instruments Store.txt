Task 303: Musical Instruments Store

class Instrument {
    private int instrumentId;
    private String brand;
    private double price;

    constructor(int id, String B, double P) {
        this.instrumentId = id;
        this.brand = B;
        this.price = P;
    }

    // Getters
    public int getInstrumentId() { return this.instrumentId; }
    public String getBrand() { return this.brand; }
    public double getPrice() { return this.price; }

    // Base method
    public String describePlayStyle() {
        return "unknown instrument";
    }
}

class Guitar extends Instrument {
    private int strings;

    constructor(int id, String B, double P, int S) {
        super(id, B, P);
        this.strings = S;
    }

    @Override
    public String describePlayStyle() {
        return "strummed or plucked";
    }
}

class DrumKit extends Instrument {
    private int pieces;

    constructor(int id, String B, double P, int Pc) {
        super(id, B, P);
        this.pieces = Pc;
    }

    @Override
    public String describePlayStyle() {
        return "struck";
    }
}

class Keyboard extends Instrument {
    private int keys;

    constructor(int id, String B, double P, int K) {
        super(id, B, P);
        this.keys = K;
    }

    @Override
    public String describePlayStyle() {
        return "pressed";
    }
}

class Store {
    private int storeId;
    private List<Instrument> instruments;

    constructor(int id) {
        this.storeId = id;
        this.instruments = new ArrayList<>();
    }

    public void addInstrument(Instrument I) {
        this.instruments.add(I);
    }

    public void showAllInstruments() {
        for (Instrument inst : this.instruments) {
            System.out.println(
                "ID: " + inst.getInstrumentId() +
                ", Brand: " + inst.getBrand() +
                ", Price: " + inst.getPrice() +
                ", Play Style: " + inst.describePlayStyle()
            );
        }
    }
}

// START
public class Main {
    public static void main(String[] args) {
        Store store1 = new Store(101);

        Guitar guitar1 = new Guitar(1, "Yamaha", 500, 6);

}

Task 304: Fitness App Workouts

class Workout {
    private int workoutId;
    private String date;
    private int duration;

    constructor(int id, String D, int Dur) {
        this.workoutId = id;
        this.date = D;
        this.duration = Dur;
    }

    // Getters
    public int getWorkoutId() { return this.workoutId; }
    public String getDate() { return this.date; }
    public int getDuration() { return this.duration; }

    public double calculateCalories() {
        return 0; // default
    }
}

class RunningWorkout extends Workout {
    private double distance;
    private double pace;

    constructor(int id, String D, int Dur, double Dist, double P) {
        super(id, D, Dur);
        this.distance = Dist;
        this.pace = P;
    }

    @Override
    public double calculateCalories() {
        return this.distance * this.pace * 0.1;
    }
}

class CyclingWorkout extends Workout {
    private double distance;
    private int resistanceLevel;

    constructor(int id, String D, int Dur, double Dist, int R) {
        super(id, D, Dur);
        this.distance = Dist;
        this.resistanceLevel = R;
    }

    @Override
    public double calculateCalories() {
        return this.distance * this.resistanceLevel * 0.2;
    }
}

class WeightTrainingWorkout extends Workout {
    private int sets;
    private int reps;

    constructor(int id, String D, int Dur, int S, int R) {
        super(id, D, Dur);
        this.sets = S;
        this.reps = R;
    }

    @Override
    public double calculateCalories() {
        return this.sets * this.reps * 0.5;
    }
}

class FitnessLog {
    private int logId;
    private List<Workout> workouts;

    constructor(int id) {
        this.logId = id;
        this.workouts = new ArrayList<>();
    }

    public void addWorkout(Workout W) {
        this.workouts.add(W);
    }

    public void showAllWorkouts() {
        for (Workout w : this.workouts) {
            System.out.println(
                "Workout ID: " + w.getWorkoutId() +
                ", Date: " + w.getDate() +
                ", Duration: " + w.getDuration() + " mins" +
                ", Calories: " + w.calculateCalories() + " calories"
            );
        }
    }

    public double calculateTotalCalories() {
        double total = 0;
        for (Workout w : this.workouts) {
            total += w.calculateCalories();
        }
        return total;
    }
}

// START
public class Main {
    public static void main(String[] args) {
        FitnessLog log1 = new FitnessLog(201);

        RunningWorkout run1 = new RunningWorkout(1, "2026-07-10", 30, 5, 6);
        CyclingWorkout cycle1 = new CyclingWorkout(2, "2026-07-11", 45, 10, 3);
        WeightTrainingWorkout wt1 = new WeightTrainingWorkout(3, "2026-07-12", 60, 4, 12);

        log1.addWorkout(run1);
        log1.addWorkout(cycle1);
        log1.addWorkout(wt1);

        System.out.println("Workouts in Fitness Log:");
        log1.showAllWorkouts();

        System.out.println("Total Calories Burned: " + log1.calculateTotalCalories() + " calories");
    }

Task 305: Video Game Characters

class Item {
    private int itemId;
    private String name;
    private double weight;
    private double value;

    constructor(int id, String N, double W, double V) {
        this.itemId = id;
        this.name = N;
        this.weight = W;
        this.value = V;
    }

    public String getName() { return this.name; }
    public double getWeight() { return this.weight; }
    public double getValue() { return this.value; }
}

class Inventory {
    private int inventoryId;
    private List<Item> items;

    constructor(int id) {
        this.inventoryId = id;
        this.items = new ArrayList<>();
    }

    public void addItem(Item item) {
        this.items.add(item);
    }

    public void showItems() {
        for (Item i : this.items) {
            System.out.println("Item: " + i.getName() + 
                               ", Weight: " + i.getWeight() + 
                               ", Value: " + i.getValue());
        }
    }
}

class Character {
    private int characterId;
    private String name;
    private int health;
    private Inventory inventory;

    constructor(int id, String N, int H) {
        this.characterId = id;
        this.name = N;
        this.health = H;
        this.inventory = new Inventory(id * 100); // link inventory ID
    }

    public String attack() {
        return this.name + " attacks!";
    }

    public void takeDamage(int amount) {
        this.health -= amount;
        if (this.health < 0) this.health = 0;
    }

    public void heal(int amount) {
        this.health += amount;
    }

    public Inventory getInventory() {
        return this.inventory;
    }
}

class Warrior extends Character {
    private String armorType;

    constructor(int id, String N, int H, String A) {
        super(id, N, H);
        this.armorType = A;
    }

    @Override
    public String attack() {
        return this.name + " attacks with melee strength!";
    }
}

class Mage extends Character {
    private int manaPool;

    constructor(int id, String N, int H, int M) {
        super(id, N, H);
        this.manaPool = M;
    }

    @Override
    public String attack() {
        if (this.manaPool > 0) {
            this.manaPool -= 10;
            return this.name + " casts a magic spell!";
        } else {
            return this.name + " has no mana left!";
        }
    }
}

class Archer extends Character {
    private int arrowCount;

    constructor(int id, String N, int H, int Ar) {
        super(id, N, H);
        this.arrowCount = Ar;
    }

    @Override
    public String attack() {
        if (this.arrowCount > 0) {
            this.arrowCount -= 1;
            return this.name + " shoots an arrow!";
        } else {
            return this.name + " has no arrows left!";
        }
    }
}

// START
public class Main {
    public static void main(String[] args) {
        Warrior warrior1 = new Warrior(1, "Thorin", 100, "Heavy Armor");
        Mage mage1 = new Mage(2, "Elora", 80, 50);

Task 306: Utility Billing System

class MeterReading {
    private int meterReadingId;
    private int startReading;
    private int endReading;

    constructor(int id, int S, int E) {
        this.meterReadingId = id;
        this.startReading = S;
        this.endReading = E;
    }

    public int getMeterReadingId() { return this.meterReadingId; }
    public int getConsumption() { return this.endReading - this.startReading; }
}

abstract class Bill {
    private int billId;
    private String billingPeriod;
    private String dueDate;
    private MeterReading meterReading;

    constructor(int id, String BP, String DD, MeterReading MR) {
        this.billId = id;
        this.billingPeriod = BP;
        this.dueDate = DD;
        this.meterReading = MR;
    }

    public int getBillId() { return this.billId; }
    public String getBillingPeriod() { return this.billingPeriod; }
    public String getDueDate() { return this.dueDate; }
    public MeterReading getMeterReading() { return this.meterReading; }

    public abstract double calculateAmountOwed();
}

class ElectricityBill extends Bill {
    private double rateDay;
    private double rateNight;
    private boolean isDayTime;

    constructor(int id, String BP, String DD, MeterReading MR, double RD, double RN, boolean dayFlag) {
        super(id, BP, DD, MR);
        this.rateDay = RD;
        this.rateNight = RN;
        this.isDayTime = dayFlag;
    }

    @Override
    public double calculateAmountOwed() {
        int consumption = this.getMeterReading().getConsumption();
        return this.isDayTime ? (consumption * this.rateDay) : (consumption * this.rateNight);
    }
}

class WaterBill extends Bill {
    private double flatRate;
    private double standingCharge;

    constructor(int id, String BP, String DD, MeterReading MR, double FR, double SC) {
        super(id, BP, DD, MR);
        this.flatRate = FR;
        this.standingCharge = SC;
    }

    @Override
    public double calculateAmountOwed() {
        int consumption = this.getMeterReading().getConsumption();
        return (consumption * this.flatRate) + this.standingCharge;
    }
}

class GasBill extends Bill {
    private double rate;
    private String season;

    constructor(int id, String BP, String DD, MeterReading MR, double R, String S) {
        super(id, BP, DD, MR);
        this.rate = R;
        this.season = S;
    }

    @Override
    public double calculateAmountOwed() {
        int consumption = this.getMeterReading().getConsumption();
        double amount = consumption * this.rate;
        if (this.season.equals("Winter")) {
            amount += 50;
        }
        return amount;
    }
}

class CustomerAccount {
    private int accountId;
    private String customerName;
    private List<Bill> bills;

    constructor(int id, String name) {
        this.accountId = id;
        this.customerName = name;
        this.bills = new ArrayList<>();
    }

    public void addBill(Bill B) {
        this.bills.add(B);
    }

    public void showAllBills() {
        System.out.println("Customer: " + this.customerName + " (Account ID: " + this.accountId + ")");
        for (Bill b : this.bills) {
            System.out.println(
                "Bill ID: " + b.getBillId() +
                ", Period: " + b.getBillingPeriod() +
                ", Due: " + b.getDueDate() +
                ", Amount: " + b.calculateAmountOwed()
            );
        }
    }
}

// START
public class Main {
    public static void main(String[] args) {
        CustomerAccount account1 = new CustomerAccount(301, "Reem");

        MeterReading mr1 = new MeterReading(101, 1000, 1200);
        ElectricityBill elecBill1 = new ElectricityBill(201, "June 2026", "2026-07-15", mr1, 0.2, 0.1, true);

        MeterReading mr2 = new MeterReading(102, 500, 650);
        WaterBill waterBill1 = new WaterBill(202, "June 2026", "2026-07-20", mr2, 0.05, 10);

        MeterReading mr3 = new MeterReading(103, 200, 300);

        account1.addBill(gasBill1);

    }
}

Task 307: Ride-Sharing Vehicle Tiers

class Route {
    private int routeId;
    private List<String> waypoints;

    constructor(int id) {
        this.routeId = id;
        this.waypoints = new ArrayList<>();
    }

    public void addWaypoint(String w) {
        this.waypoints.add(w);
    }

    public void showRoute() {
        System.out.println("Route ID: " + this.routeId);
        for (String wp : this.waypoints) {
            System.out.println("Waypoint: " + wp);
        }
    }

    // Stubbed distance calculation (could be improved with coordinates)
    public int calculateDistance() {
        return this.waypoints.size() * 5; // assume 5 km per segment
    }
}

abstract class Ride {
    private int rideId;
    private String pickupLocation;
    private String dropoffLocation;
    private Route route;
    private int distance;

    constructor(int id, String P, String D, int Dist, Route RouteObj) {
        this.rideId = id;
        this.pickupLocation = P;
        this.dropoffLocation = D;
        this.distance = Dist;
        this.route = RouteObj;
    }

    public int getRideId() { return this.rideId; }
    public Route getRoute() { return this.route; }

    public abstract double calculateFare();
}

class StandardRide extends Ride {
    private double ratePerKm;

    constructor(int id, String P, String D, int Dist, Route RouteObj, double Rate) {
        super(id, P, D, Dist, RouteObj);
        this.ratePerKm = Rate;
    }

    @Override
    public double calculateFare() {
        return this.getRoute().calculateDistance() * this.ratePerKm;
    }
}

class SharedRide extends StandardRide {
    private int passengerCount;

    constructor(int id, String P, String D, int Dist, Route RouteObj, double Rate, int PC) {
        super(id, P, D, Dist, RouteObj, Rate);
        this.passengerCount = PC;
    }

    @Override
    public double calculateFare() {
        double baseFare = super.calculateFare();
        return baseFare / this.passengerCount;
    }
}

class PremiumRide extends Ride {
    private double ratePerKm;
    private double comfortSurcharge;

    constructor(int id, String P, String D, int Dist, Route RouteObj, double Rate, double CS) {
        super(id, P, D, Dist, RouteObj);
        this.ratePerKm = Rate;
        this.comfortSurcharge = CS;
    }

    @Override
    public double calculateFare() {
        return (this.getRoute().calculateDistance() * this.ratePerKm) + this.comfortSurcharge;
    }
}

// START
public class Main {
    public static void main(String[] args) {
        Route route1 = new Route(501);
        route1.addWaypoint("Mall");
        route1.addWaypoint("Main Street");
        route1.addWaypoint("Airport");

        System.out.println("Route followed:");
        route1.showRoute();

        StandardRide standardRide1 = new StandardRide(101, "Mall", "Airport", 15, route1, 0.5);
        SharedRide sharedRide1 = new SharedRide(102, "Mall", "Airport", 15, route1, 0.5, 3);
        PremiumRide premiumRide1 = new PremiumRide(103, "Mall", "Airport", 15, route1, 1.0, 5);

        System.out.println("Standard Ride Fare: " + standardRide1.calculateFare());
        System.out.println("Shared Ride Fare: " + sharedRide1.calculateFare());
        System.out.println("Premium Ride Fare: " + premiumRide1.calculateFare());
    }
}
