class attempt
attemptId: number
exam: exam
answers[]
score: number
date: text
timeLimit: number
expired: boolean

constructor(AI: number , E: exam , D: text , TL: number)
set attemptId to AI
set exam to E
set date to D
set timeLimit to TL
set answers to []
set score to 0
set expired to false
end constructor

method: answerQuestion(Q: question , ANS: text)
if expired = false
add (Q, ANS) to answers
 else
 display "Time limit passed, cannot answer."
end if
    end method

 method: calculateScore()
 set score to 0
for each (Q, ANS) in answers
if ANS = Q.correctAnswer
score = score + Q.pointValue
end if
end for
end method

method: expireAttempt()
set expired to true
end method
end class
