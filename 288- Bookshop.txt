class Question
    private text: text
    private correctAnswer: text
    private pointValue: number

    constructor(TX: text , CA: text , PV: number)
        set this.text to TX
        set this.correctAnswer to CA
        set this.pointValue to PV
    end constructor
end class


class Attempt
    private attemptId: number
    private exam: Exam
    private answers[]   // list of pairs (Question, Answer)
    private score: number
    private date: text
    private timeLimit: number
    private expired: boolean
    private startTime: text

    constructor(AI: number , E: Exam , D: text , TL: number)
        set this.attemptId to AI
        set this.exam to E
        set this.date to D
        set this.timeLimit to TL
        set this.answers to []
        set this.score to 0
        set this.expired to false
        set this.startTime to D
    end constructor

    method: answerQuestion(Q: Question , ANS: text , currentTime: text)
        if this.expired = false AND currentTime <= this.startTime + this.timeLimit
            add (Q, ANS) to this.answers
        else
            display "Time limit passed, cannot answer."
            set this.expired to true
        end if
    end method

    method: calculateScore()
        set this.score to 0
        for each (Q, ANS) in this.answers
            if ANS = Q.correctAnswer
                this.score = this.score + Q.pointValue
            end if
        end for
    end method

    method: expireAttempt()
        set this.expired to true
    end method
end class

class Member
    private name: text
    private contactInfo: text
    private membership: Membership
    private bookings[]

    constructor(NM: text , CI: text , M: Membership)
        set this.name to NM
        set this.contactInfo to CI
        set this.membership to M
        set this.bookings to []
    end constructor

    method: bookRoom(BI: text , R: Room , D: text , ST: text , ET: text)
        duration = calculateHours(ST, ET)
        if this.getUsedHours() + duration <= this.membership.monthlyQuota
            if R.isAvailable(D, ST, ET)
                create object B of Booking(BI, D, ST, ET, this, R)
                add B to this.bookings
                R.addBooking(B)
                display "Booking successful"
            else
                display "Room not available in this time slot."
            end if
        else
            display "Booking rejected: quota exceeded."
        end if
    end method

    method: getUsedHours()
        total = 0
        for each booking in this.bookings
            total = total + calculateHours(booking.startTime, booking.endTime)
        end for
        return total
    end method

    method: getRemainingHours()
        return this.membership.monthlyQuota - this.getUsedHours()
    end method
end class


class Membership
    private membershipNo: number

class Room
    private roomNo: number
    private bookings[]

    constructor(RN: number)
        set this.roomNo to RN
        set this.bookings to []
    end constructor

    method: isAvailable(D: text , ST: text , ET: text)
        for each booking in this.bookings
            if timesOverlap(ST, ET, booking.startTime, booking.endTime) AND booking.date = D
                return false
            end if
        end for
        return true
    end method

    method: addBooking(B: Booking)
        add B to this.bookings
    end method
end class
