Task 293: Shape Drawing App

class Shape
    private color: text
    private positionX: number
    private positionY: number

    constructor(C: text , X: number , Y: number)
        set this.color to C
        set this.positionX to X
        set this.positionY to Y
    end constructor

    method: getArea()
       
    end method

    method: getPerimeter()
       
    end method
end class


class Circle inherits Shape
    private radius: number

    constructor(C: text , X: number , Y: number , R: number)
        call super(C, X, Y)
        set this.radius to R
    end constructor

    method: getArea()
        return 3.14 * this.radius * this.radius
    end method

    method: getPerimeter()
        return 2 * 3.14 * this.radius
    end method
end class


class Rectangle inherits Shape
    private width: number
    private height: number

    constructor(C: text , X: number , Y: number , W: number , H: number)
        call super(C, X, Y)
        set this.width to W
        set this.height to H
    end constructor

    method: getArea()
        return this.width * this.height
    end method

    method: getPerimeter()
        return 2 * (this.width + this.height)
    end method
end class


class Triangle inherits Shape
    private sideA: number
    private sideB: number
    private sideC: number

    constructor(C: text , X: number , Y: number , A: number , B: number , Cc: number)
        call super(C, X, Y)
        set this.sideA to A
        set this.sideB to B
        set this.sideC to Cc
    end constructor

    method: getArea()
        s = (this.sideA + this.sideB + this.sideC) / 2
        return sqrt(s * (s - this.sideA) * (s - this.sideB) * (s - this.sideC))
    end method

    method: getPerimeter()
        return this.sideA + this.sideB + this.sideC
    end method
end class


class Canvas
    private shapes[]

    constructor()
        set this.shapes to []
    end constructor

    method: addShape(S: Shape)
        add S to this.shapes
    end method

    method: showAllShapes()
        for each shape in this.shapes
            display shape.getArea() , shape.getPerimeter()
        end for
    end method
end class


START
BEGIN
    CREATE OBJECT canvas1 OF Canvas()

    CREATE OBJECT circle1 OF Circle("Red", 10, 20, 5)
    CREATE OBJECT rect1 OF Rectangle("Blue", 15, 25, 10, 20)
    CREATE OBJECT tri1 OF Triangle("Green", 5, 5, 3, 4, 5)

    canvas1.addShape(circle1)
    canvas1.addShape(rect1)
    canvas1.addShape(tri1)

    display "Shapes on Canvas:"
    canvas1.showAllShapes()
END
STOP
